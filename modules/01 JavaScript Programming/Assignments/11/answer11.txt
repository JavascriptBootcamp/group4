var i = 1;
var str = "";

while (i < 51) {
    str = i.toString();
    if ((str.search("7") != -1) || i % 7 === 0)
        console.log(i + " boom");
    i++;
}

i = 1;
while (i < 101) {
    if (i % 3 === 0 && i % 5 === 0)
        console.log(i + " FizzBuzz");
    else if (i % 5 === 0)
        console.log(i + " buzz");
    else if (i % 3 === 0)
        console.log(i + " fizz");

    i++;
}
