//Write a function that given a string of words, 
//returns a string composed of the even words (1-based) from the original string:

function evenWords(str) {
  var answer = str.split(" ").filter(function (item) {
      return item.length%2===0;
  }).join(" ");
  
  return answer;
}

var printAnswer = evenWords("aaa bbbb asda 1331 ewe");
console.log(printAnswer);
