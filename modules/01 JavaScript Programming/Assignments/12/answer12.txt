function evenWords(str) {

    arr = str.split(" ");
    var filtered = arr.filter(function(element, index) {
        
            if((index+1)%2===0)
                return element;
        })
    var newStr = filtered.join(' ');
    console.log(newStr);
}

myStr ='aaa bbbb asda 1331 ewe';
