var str = "aaa bbbb asda 1331 ewe";

console.log(evenWords(str));

function evenWords(str) {
    var arr = str.split(" ");
    var newstr = "";
    var count = 1;

    for (var i in arr) {
        if (count % 2 === 0) {
            newstr += arr[i];
            if (i !== (arr.length - 1))
                newstr += " ";
        }
        count++;
    }

    return newstr;
}
