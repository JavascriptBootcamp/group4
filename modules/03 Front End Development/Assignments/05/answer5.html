            var arr = [1, 2, 5, 7];
            function myFilter(myArr, callback){
            var result =[];
            for(var i=0 ; i < myArr.length ; i++){
                    if(callback(myArr[i]))
                        result.push(myArr[i]);
            }
            return result;
            }
            var filtered = myFilter(arr, function(item){
                return item % 2 === 0;
