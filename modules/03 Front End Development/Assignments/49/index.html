            var myArr = [1, 2, 5, 7];
            function myFilter(arr, callback){
            var result = [];
            for(var i=0 ; i < arr.length ; i++){
                    if(callback(arr[i]))
                    result.push(arr[i]);
            }
            return result;
            }
            var filtered = myFilter(myArr, function(item){
                return item % 2 === 0;
            });
            console.log(filtered);
